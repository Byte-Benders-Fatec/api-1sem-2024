CREATE TABLE aluno (
	id INT(10) NOT NULL,
	cpf VARCHAR(14) NOT NULL,
	nome VARCHAR(255) NOT NULL,
	data_nascimento DATE(10) NOT NULL,
	peso FLOAT(12) NOT NULL,
	altura FLOAT(12) NOT NULL,
	PRIMARY KEY (id),
);

CREATE TABLE historico (
	id INT(10) NOT NULL,
	aluno_id INT(10) NOT NULL,
	data_registro DATE(10) NOT NULL,
	peso FLOAT(12) NOT NULL,
	PRIMARY KEY (id),
	FOREIGN KEY (aluno_id) REFERENCES aluno(id),
);
